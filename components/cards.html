Cards

Cards

Cards provide a flexible and extensible content container with multiple variants and options

💡

Check out our ladle for more info on this component.

Example

import {Card} from '@comfortdelgro/react-compass'
 
function BasicExample() {
  return (
    <Card size='sm'>
      <Card.Body>
        <Card.Title>Card title</Card.Title>
        <p>This is a card paragraph</p>
      </Card.Body>
    </Card>
  )
}

Content

Body

Use <Card.Body> to pad content inside a <Card>.

import {Card} from '@comfortdelgro/react-compass'
 
function BodyOnlyExample() {
  return (
    <Card size='sm'>
      <Card.Body>
        <p>This is a paragraph within a card body</p>
      </Card.Body>
    </Card>
  )
}

Title

Use <Card.Title> to create a title inside a <Card.Body>.

import {Card} from '@comfortdelgro/react-compass'
 
function TitleExample() {
  return (
    <Card size='sm'>
      <Card.Body>
        <Card.Title>Card title</Card.Title>
        <p>This is a paragraph within a card body</p>
      </Card.Body>
    </Card>
  )
}

Action

Use <Card.Action> to create an area for action buttons in a card <Card.Body>.

import {Card} from '@comfortdelgro/react-compass'
 
function CardActionExample() {
  return (
    <Card size='lg'>
      <Card.Body>
        <Card.Title>Card title as H1</Card.Title>
        <p>This is a paragraph within a card body</p>
        <Card.Action>
          <Button className='c-hKssGC-hYGuEl-variant-primary' variant='primary'>Button</Button>
          <Button variant='secondary'>Button</Button>
        </Card.Action>
      </Card.Body>
    </Card>
  )
}

Image

Use <Card.Image> to embed the image in a card <Card>.

import {Card} from '@comfortdelgro/react-compass'
 
function ImageExample() {
  const imageSrc =
    'https://images.unsplash.com/flagged/photo-1562503542-2a1e6f03b16b?ixlib=rb-4.0.3&ixid=MnwxMjA3fDB8MHxzZWFyY2h8NXx8c2luZ2Fwb3JlfGVufDB8fDB8fA%3D%3D&w=1000&q=80'
 
  return (
    <Card size='lg'>
      <Card.Image src={imageSrc} />
      <Card.Body>
        <Card.Title>Card with image</Card.Title>
        <p>This is a paragraph within a card body</p>
      </Card.Body>
    </Card>
  )
}

Sizes

Card comes with different sizes: sm, lg and full.

import {Card} from '@comfortdelgro/react-compass'
 
function SizesExample() {
    const sizes = ['sm', 'lg', 'full']
 
    return (
       {sizes.map((size) => (
        <Card size={size}>
          <Card.Body>
            <Card.Title>
              Card with size: {size}
            </Card.Title>
            <p>This is a paragraph within a card body</p>
          </Card.Body>
        </Card>
       ))}
    );
}

      </Card.Body>
    </Card>

Disabled

Card can be disabled by passing isDisabled prop.

import {Card} from '@comfortdelgro/react-compass'
import Button from '@comfortdelgro/react-compass/button'
 
function DisabledCard() {
  const isDisabled = true
  return (
    <Card size='sm' isDisabled={isDisabled}>
      <Card.Body>
        <Card.Title>Disabled Card</Card.Title>
        <p>This is a paragraph within a card body</p>
        <Card.Action>
          <Button isDisabled={isDisabled} variant='primary'>
            Button
          </Button>
        </Card.Action>
      </Card.Body>
    </Card>
  )
}

Clickable

Card can be clicked and focused by passing isClickable prop.

import {Card} from '@comfortdelgro/react-compass'
 
function DisabledCard() {
  return (
    <Card size='sm' isClickable onClick={() => alert('clicked')}>
      <Card.Body>
        <Card.Title>Clickable Card</Card.Title>
        <p>This is a paragraph within a card body</p>
      </Card.Body>
    </Card>
  )
}

Shadowless

Card will not have shadows when passing isShadowless prop.

import {Card} from '@comfortdelgro/react-compass'
 
function DisabledCard() {
  return (
    <Card size='sm' isShadowless>
      <Card.Body>
        <Card.Title>Shadowless Card</Card.Title>
        <p>This is a paragraph within a card body</p>
      </Card.Body>
    </Card>
  )
}
